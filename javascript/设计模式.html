# 工厂模式

# 特点: 注重结果，过程直接交给工厂，我只在乎返回的结果
function Person(name) {
    this.name = name;
    return this;
}
Person.prototype.writeJs = function () {
        console.log(this.name + 'write js');
}
var a = new Person('xiaobai');
a.writeJs(); // xiaobaiwrite js

var b = new Person('xiaobai');
b.writeJs(); // xiaobaiwrite js

console.log(a === b)  // false

但是，很奇怪耶，明明new出来的都是xiaobai，结果 a不等于b了

其实，以上代码每一次new新对象的时候，都会分配一个新的内存空间，因此a的内存位置不等于b，所以就会返回false。

这个时候，我们想到的，就是应该如果做到让a === b。

请看以下代码

function Person(name) {
    if (Person.instance instanceof  Object){
        return Person.instance
    }
    this.name = name
    Person.instance = this
    return this
}
Person.prototype.writeJs = function () {
        console.log(this.name + 'write js');
}
var a = new Person('xiaobai');
a.writeJs(); // xiaobaiwrite js

var b = new Person('xiaobai');
b.writeJs(); // xiaobaiwrite js

console.log(a === b)  // true

  1. 在Person类上定义一个instance,用于记录这个实例化后的对象，第一次实例化前 instance === undefined ，

  2. 实例化后，instance变成了实例化后的对象。

  3. 第二次实例化时候，会先判断instance是否已经是object了，如果是，就直接返回它，可以避免一个对象被重复实例化~，酱紫就ok了

    # 以上就是单例工厂模式！！！！

# 策略模式

# 特点: 策略模式就是为了避免那些重复if,else的出现,实现代码复用

例子:

function getSalary = function (who, level){
    let salary = 0
    switch(level){
        case 1:
            salary =  level *1200
            break
        case 2:
            salary =  level *1200
            break
        case 3:
            salary =  level *1200
            break
    }
    console.log(who + 'can get ' + salary)
}
getSalary('lisan',1) // lisancan get 12k
getSalary('张四',2)  // 张四can get 13k
getSalary('老五',3)  // 老五can get 13k

以上代码出现了过多次数的case判断，浪费内存，可以采取以下的包含在对象里的方法，实现节约内存

let controller = {
	1: function(who){
		console.log(who+'can get 12k ')
	},
	2: function(who){
		console.log(who+'can get 13k ')
	},
	3: function(who){
		console.log(who+'can get 14k ')
	}
}
let getSalary = function (who,level){
	controller[level](who);
}
getSalary('lisan',1) // lisancan get 12k
getSalary('张四',2)  // 张四can get 13k
getSalary('老五',3)  // 老五can get 13k
