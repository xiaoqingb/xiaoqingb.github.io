# SelectSort

let arr = [2,1,9,5,4,2,10,66,77]
function swap(arr,index1,index2){
    let temp = arr[index1]
    arr[index1] = arr[index2]
    arr[index2] = temp
}
function SelectSort(array){
    let length = array.length;
    for(let i = 0 ; i < length - 1; i++){
        let minIndex = i
        for(let j = i + 1 ; j < length ; j++){
            if(array[minIndex] > array[j]){
                minIndex = j
            }
        }
        // 如果没有比arr[i]小的，则继续循环
        if(i === minIndex) continue;
        swap(arr,i,minIndex)
    }
    return array
}
console.log(SelectSort(arr))

function quickSort(arr, left, right){
        let baseVal=arr[left]
        while(left < right){
            while(left < right && arr[right] >= baseVal){
                right--
            }
            swap(arr, left, right)
            while(left < right && arr[left] <= baseVal){
                left++
            }
            swap(arr, left, right)
        }
        return left

# insertSort

 function insertionSort (arr,n) {
        let key
        for (let i = 1 ; i < n; i++) {
            // 选中的值
            key = arr[i]
            j = i - 1
            // 将选中的值和前面的值比较
            // 如果前面的值比选中的值大，则前面的值一直往后挪
            // 直到没有比它大的，停止循环，并在j+1处赋值
            while(j >= 0 && arr[j] > key ){
                console.log(arr)  //这里的输出可以清晰的看出数据移动的变化
                arr[j+1] = arr[j]
                --j
            }
            arr[j+1] = key
            console.log(arr)
        }
        return arr
    }
    console.log(insertionSort(arr))

# bubbleSort

function BubbleSort(array){
    let length=array.length;
    for (let i = 0;i < length - 1; i++) {
        for (let j = 0; j <length - i; j++){
            if(array[j]<array[j + 1]) {
                let temp = array[j];
                array[j] = array[j+1];
                array[j + 1] = temp;
            }
        }
    }
    return array
}
